    <script>
        // Compliment Slider Logic
        const compliments = [
            "Your kindness is a gift to everyone who meets you.",
            "You are incredibly resilient, even when you don't feel like it.",
            "The world is a much better place because you're in it.",
            "You have the most wonderful perspective on things.",
            "Being around you feels like a warm cup of tea on a rainy day.",
            "You are enough, exactly as you are right now.",
            "Your laugh is genuinely my favorite sound."
        ];

        let compIndex = 0;
        const compElement = document.getElementById('compliment-text');

        function rotateCompliments() {
            compElement.style.opacity = 0;
            setTimeout(() => {
                compIndex = (compIndex + 1) % compliments.length;
                compElement.innerText = compliments[compIndex];
                compElement.style.opacity = 1;
            }, 500);
        }
        setInterval(rotateCompliments, 4000);

        // Smile Generator Logic
        const smileContent = [
            "Joke: What do you call a bear with no teeth? A gummy bear! 🧸",
            "Quote: 'Believe you can and you're halfway there.' - Theodore Roosevelt",
            "Reminder: You've survived 100% of your hardest days so far. You're doing great.",
            "Fact: Otters hold hands when they sleep so they don't drift apart. 🦦",
            "Quote: 'You are braver than you believe, stronger than you seem, and smarter than you think.'",
            "Joke: Why don't scientists trust atoms? Because they make up everything! ⚛️",
            "Message: Sending you a giant cup of virtual coffee and a huge hug. ☕"
        ];

        function generateSmile() {
            const display = document.getElementById('smile-display');
            const random = smileContent[Math.floor(Math.random() * smileContent.length)];
            display.style.opacity = 0;
            setTimeout(() => {
                display.innerText = random;
                display.style.opacity = 1;
            }, 200);
        }

        // Floating Hearts Logic
        function createHeart() {
            const heart = document.createElement('div');
            heart.classList.add('heart');
            heart.innerHTML = '❤️';
            heart.style.left = Math.random() * 100 + "vw";
            heart.style.animationDuration = Math.random() * 3 + 3 + "s";
            document.body.appendChild(heart);

            setTimeout(() => {
                heart.remove();
            }, 6000);
        }

        setInterval(createHeart, 500);
    </script>
